def dyn(u0, y0, f, t, alpha=1):
    dt = t[1]-t[0]
    u = u0
    y = y0
    U = []
    Y = []
    for t_ in t:
        y = y + dt*f(alpha*u,y)
        u = u + dt*y
        U += [u]
        Y += [y]
    return U, Y

def natural_freq(time, proto_phases):
    try:
        r, c = proto_phases.shape
        ax = 1
        if r > c:
            ax = 0
    except ValueError:
        ax = 0
    proto_phases = np.unwrap(proto_phases)
    freq = (proto_phases[-1] - proto_phases[0])/(time[-1] - time[0])
    return freq # omega_0

    for t_ in t:
